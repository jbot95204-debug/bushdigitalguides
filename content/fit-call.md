(function(){
  var form = document.getElementById('bdg-fit-call-form');
  var status = document.getElementById('bdg-fit-form-status');
  if (!form || !status) return;
  var started = Date.now();
  form.addEventListener('submit', function(event){
    event.preventDefault();
    status.textContent = '';
    var data = new FormData(form);
    if ((data.get('company_website') || '').trim()) {
      status.textContent = 'Thanks. Please email hello@bushdigitalguides.com.au directly if the form does not open.';
      return;
    }
    if (Date.now() - started < 2500) {
      status.textContent = 'Please check the details, then press send again.';
      started = Date.now() - 2500;
      return;
    }
    var required = ['name','business','email','stuck'];
    for (var i = 0; i < required.length; i++) {
      if (!String(data.get(required[i]) || '').trim()) {
        status.textContent = 'Please fill in the required fields first.';
        return;
      }
    }
    var subject = 'Fit Call request — ' + String(data.get('business') || '').trim();
    var body = [
      'Name: ' + String(data.get('name') || '').trim(),
      'Business name: ' + String(data.get('business') || '').trim(),
      'Email: ' + String(data.get('email') || '').trim(),
      'Phone: ' + String(data.get('phone') || '').trim(),
      'Website or Google-profile link: ' + String(data.get('link') || '').trim(),
      'Team size: ' + String(data.get('team') || '').trim(),
      'Main service area: ' + String(data.get('area') || '').trim(),
      'What is currently getting missed, delayed or repeated?: ' + String(data.get('stuck') || '').trim(),
      'What would a good result look like?: ' + String(data.get('result') || '').trim(),
      'Preferred contact method: ' + String(data.get('method') || '').trim(),
      '',
      'Safety reminder: I have not included passwords, card or bank details, login links, private customer lists, or urgent legal/compliance matters.'
    ].join('\n');
    window.location.href = 'mailto:hello@bushdigitalguides.com.au?subject=' + encodeURIComponent(subject) + '&body=' + encodeURIComponent(body);
    status.textContent = 'Your email app should open now. Review the message, then press send if you want to request a Fit Call.';
  });
})();
</script>
